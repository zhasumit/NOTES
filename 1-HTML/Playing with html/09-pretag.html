            {
                for (int i = 0; i < size - 1; i++)
                {
                    int minIndex = i;
                    // find the minimum
                    for (int j = i + 1; j < size; j++)
                        if (arr[j] < arr[minIndex])
                            minIndex = j;

                    // swap the elements minimum <-> ith index
                    int temp = arr[minIndex];
